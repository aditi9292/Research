

%%% Get the amount of times each permission has been accepted or rejected


select Count(ar.PermissionName) as TotalPermission, ar.PermissionName, NoRat.PermCount as TotalNoRat, Rat.PermCount as TotalRat, RatAccept.PermCount as RatAccept, RatReject.PermCount as RatReject, NoRatAccept.PermCount as NoRatAccept, NoRatReject.PermCount as NoRatReject
from appresults ar
-- Get the total number of times the permission was requested with No Rational
inner join (
	select Count(NoRat.PermissionName) as PermCount, NoRat.PermissionName 
	from appresults NoRat
	where TestType = 'No Rational'
	group by NoRat.PermissionName
) NoRat on NoRat.PermissionName = ar.PermissionName

-- Get the total number of times the permission was requested with Rational
inner join (
	select Count(Rat.PermissionName) as PermCount, Rat.PermissionName 
	from appresults Rat
	where TestType = 'Rational'
	group by Rat.PermissionName
) Rat on Rat.PermissionName = ar.PermissionName

-- Rat Accept
inner join (
	select Count(NoRat.PermissionName) as PermCount, NoRat.PermissionName
	from appresults NoRat
	where TestType = 'Rational'
	and isGranted = 'Grant'
	group by NoRat.PermissionName
) RatAccept on RatAccept.PermissionName = ar.PermissionName

-- Rat Reject
inner join (
	select Count(NoRat.PermissionName) as PermCount, NoRat.PermissionName
	from appresults NoRat
	where TestType = 'Rational'
	and isGranted = 'Deny'
	group by NoRat.PermissionName
) RatReject on RatReject.PermissionName = ar.PermissionName


-- No Rat Accept
inner join (
	select Count(NoRat.PermissionName) as PermCount, NoRat.PermissionName
	from appresults NoRat
	where TestType = 'No Rational'
	and isGranted = 'Grant'
	group by NoRat.PermissionName
) NoRatAccept on NoRatAccept.PermissionName = ar.PermissionName


-- No Rat Reject
inner join (
	select Count(NoRat.PermissionName) as PermCount, NoRat.PermissionName
	from appresults NoRat
	where TestType = 'No Rational'
	and isGranted = 'Deny'
	group by NoRat.PermissionName
) NoRatReject on NoRatReject.PermissionName = ar.PermissionName



group by ar.PermissionName
