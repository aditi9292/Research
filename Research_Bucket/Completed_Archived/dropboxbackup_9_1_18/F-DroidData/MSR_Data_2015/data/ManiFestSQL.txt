-- Most commonly used permissions
select amp.Permission, count (amj.permission_ID) as permissionIDCount
from Android_Manifest_Permission_join amj
inner join android_manifest_permission amp on amj.Permission_ID = amp.Permission_ID
group by amj.permission_ID
order by permissionIDCount desc
